function authLogout() {
  var usuario = AUTH.currentUser ? AUTH.currentUser.usuario : '—';
  authLog("LOGOUT", "Saiu do sistema");
  tickerAdd({ tipo:'LOGOUT', usuario: usuario });
  
  // ════ LIMPEZA COMPLETA DO ESTADO ════
  AUTH.currentUser = null;
  localStorage.clear();                    // ← NOVO
  sessionStorage.clear();                  // ← NOVO
  
  try {
    if (window.indexedDB) {
      indexedDB.databases().then(function(dbs) {
        dbs.forEach(function(db) {
          indexedDB.deleteDatabase(db.name); // ← NOVO
        });
      }).catch(function(e) {
        console.warn('[LOGOUT] Erro:', e);
      });
    }
  } catch(e) {}
  
  if (window.AUTH) {
    AUTH.permissions = {};                 // ← NOVO
    AUTH.userRole = null;                  // ← NOVO
    AUTH.userSectors = [];                 // ← NOVO
    AUTH.userShifts = [];                  // ← NOVO
  }
  
  if ('serviceWorker' in navigator) {
    navigator.serviceWorker.getRegistrations().then(function(regs) {
      regs.forEach(function(reg) {
        reg.unregister().catch(function(e) {}); // ← NOVO
      });
    }).catch(function(e) {});
  }
  
  setTimeout(function() {                  // ← NOVO
    location.reload();
  }, 100);
}
